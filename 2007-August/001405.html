+import numpy
+
+def getnan(a):
+    """
+    emulate numarray.ieeespecial.getnan on numpy arrays
+    """
+    return numpy.nonzero(numpy.isnan(a))
+
+def stddev(a):
+    import math
+    n= a.size
+    if n==0: return nan
+    if n==1: return 0
+    return math.sqrt(((a - a.mean())**2).sum()/(n-1))
