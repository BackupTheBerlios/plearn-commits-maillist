+bool isemptyFile(const PPath& path)
+{
+    PRFileInfo64 fi;
+
+    if (PR_GetFileInfo64(path.absolute().c_str(), &fi) != PR_SUCCESS)
+        return false;
+    else
+        return (fi.type == PR_FILE_FILE) && (fi.size == 0);
